int converte(char string[])
{
	int i,tamnum,j;
	tamnum=strlen(string);
	
	for(i=0;i<tamnum;i++)
	{
		string[i]=tolower(string[i]);
		string[i+1]=toupper(string[i+1]);
		i++;
	}
	
	
	printf("string=%s",string);
}
